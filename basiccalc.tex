\section{Basic Calculus}
\subsection{Differentiation}
\textbf{Rule - Differentiation of Powers}
\begin{equation}
\begin{split}
y = x^{n} \\
\frac{dy}{dx} = nx^{n-1}
\end{split}
\end{equation}




\textbf{Example 1}
$y = x^{2}\\
\frac{dy}{dx} = 2x^{2-1} = 2x
$


\textbf{Example 2}
$y = x^{2}\\
\frac{dy}{dx} = 2x^{2-1} = 2x
$


\textbf{Example 3}
$y = x\\
\frac{dy}{dx} = 1x^{1-1} = 1.1
$

\textbf{Example 4}
$y = 3\\
\frac{dy}{dx} = 3.0 = 0
$

\textbf{Example 5}
$y = 4x^{3}\\
\frac{dy}{dx} = 4*3*x^{3-1} = 12x^{2}
$


\textbf{Example 6}
$y = 3x + 3x^{4} + 7\\
\frac{dy}{dx} = 3*x^{1-1} + 3*4*x^{4-1} + 7 * 0 \\
\frac{dy}{dx} = 3*x^{0} + 12*x^{3} + 0\\
\frac{dy}{dx} = 3 + 12x^{3} \\
$
